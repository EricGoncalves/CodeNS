MACRO(ADD_FILE_DEPENDANCY file)
   GET_SOURCE_FILE_PROPERTY( old_dep ${file} 
      OBJECT_DEPENDS ) 
   if ( old_dep STREQUAL "NOTFOUND")
     SET(${file}_deps ${ARGN}) 
   else()
     SET(${file}_deps ${old_dep} ${ARGN}) 
   endif()
   SET_SOURCE_FILES_PROPERTIES( ${file} 
      PROPERTIES OBJECT_DEPENDS "${${file}_deps}" ) 
ENDMACRO(ADD_FILE_DEPENDANCY)
